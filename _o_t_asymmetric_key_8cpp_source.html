01356             packetTag &= 0x3F;

01359                 packetLength = ((packetLength-192) << 8) + pbData[i++];
01360             else if( (packetLength > 223) && (packetLength < 255))
01361                 packetLength = (1 << (packetLength & 0x1f));

01365                 i+=4;

01368         else
01369         {
01370             packetLength = packetTag & 3;

01372             if(packetLength == 0) 
01373                 packetLength = pbData[i++];
01374             else if(packetLength == 1) 
01375             {
01376                 packetLength = (pbData[i]<<8) + pbData[i+1];
01377                 i+=2;
01378             }

01380             {
01381                 packetLength = (pbData[i]<<24) + (pbData[i+1]<<16) + (pbData[i+2]<<8) + pbData[i+3];
01382                 i+=4;

01384             else 
01385                 packetLength = dataLength - 1;
01386         }
01387
