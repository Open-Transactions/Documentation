00239     OT_ASSERT(theInput.Exists());

00241     // Generate a random initialization vector.
00242     //
00243     OTPayload theIV;
00244

00252     // (The passphrase is used to derive another key that is used to encrypt the

00281     //

00283     
00284     const bool bEncrypted = OTCrypto::It()->Encrypt(theRawSymmetricKey, // The symmetric key, in clear form.

00287                                                     theInput.GetLength() + 1, // for null terminator
00288                                                     // -------------------------------

00295     //

00329     
00330     m_dataContents.Concatenate(reinterpret_cast<void *>(&ivlen_n),   
00331                                static_cast<uint32_t>(sizeof(ivlen_n)));

00334     //

00344     // We don't write the size of the ciphertext before the ciphertext itself,
00345     // since the decryption is able to deduce the size based on the total envelope
00346     // size minus the other pieces. We might still want to add that size here, however.
00347     // (for security / safety reasons.)
00348
