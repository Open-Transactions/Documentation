00555              on client side, and only after the ones with actual replies (as evidenced
00556              by the Nymbox) have already been removed from *this "sent buffer."
00557              
00558              Why were they removed in advance? Because clearly: if the server HAS replied
00559              to them already, then there's no need to harvest anything: just let it
00560              process as normal, whether the transaction inside is a success or fail.
00561              (We KNOW the message didn't fail because otherwise there wouldn't even be
00562              a notice in the Nymbox. So this is about the transaction inside.)
00563 
00564              So we remove the ones that we DEFINITELY know the server HAS replied to.
00565              
00566              And the ones remaining? We know for those, the server definitely has NOT
00567              replied to them (the message must have been dropped by the network or
00568              something.) How do we know this? Because there would be a notice in the
00569              Nymbox! So at the moment of successful @getNymbox, we are able to loop through
00570              those receipts and know FOR SURE, WHICH ones definitely have a reply, and
00571              which ones definitely DO NOT.
00572              
00573              The ones where we definitely do NOT have a reply--that is, the ones that are in
00574              the "sent messages" buffer, but are not in the Nymbox with the same request
00575              number--we harvest those numbers, since the server clearly never saw them, or
00576              rejected the message before the transaction itself even had a chance to run.
00577              
00578              */
00579             if (NULL != pNym)
00580             {
00581                 OT_ASSERT(NULL != pstrNymID && pstrNymID->Exists());
00582                 const OTIdentifier MSG_NYM_ID(*pstrNymID);
00583                 OT_ASSERT(pNym->CompareID(MSG_NYM_ID));
00584                 // ----------------------------
00585                 OT_ASSERT(NULL != pstrServerID && pstrServerID->Exists());
00586                 // ----------------------------
00587                 OT_ASSERT(NULL != pbHarvestingForRetry);
00588                 // ----------------------------
00589                 /*
00590                  getNymbox          -- client is NOT sending hash, server is NOT rejecting bad hashes, server IS SENDING HASH in the @getNymbox reply
00591                  getRequest         -- client is NOT sending hash, server is NOT rejecting bad hashes, server IS SENDING HASH in the @getRequest reply
00592                  
00593                  processNymbox      -- client is SENDING HASH, server is REJECTING BAD HASHES, server is SENDING HASH in the @processNymbox  reply
00594                  notarizeTransactions   -- client is SENDING HASH, server is REJECTING BAD HASHES, server is SENDING HASH in the @notarizeTransactions  reply
00595                  processInbox       -- client is SENDING HASH, server is REJECTING BAD HASHES, server is SENDING HASH in the @processInbox  reply
00596                  triggerClause      -- client is SENDING HASH, server is REJECTING BAD HASHES, server is SENDING HASH in the @triggerClause reply
00597                  
00598                  getTransactionNum  -- client is SENDING HASH, server is REJECTING BAD HASHES, server is SENDING HASH in the @getTransactionNum reply
00599                  
00600                  Already covered in NotarizeTransaction: 
00601                     transfer, withdrawal, deposit, marketOffer, paymentPlan, smartContract, cancelCronItem, exchangeBasket
00602                  */
00603                 
00604                 if (pThisMsg->m_ascPayload.Exists() &&
00605                     (
00606                      pThisMsg->m_strCommand.Compare("processNymbox")        ||
00607                      pThisMsg->m_strCommand.Compare("processInbox")         ||
00608                      pThisMsg->m_strCommand.Compare("notarizeTransactions") ||
00609                      pThisMsg->m_strCommand.Compare("triggerClause")
00610                     )
00611                    )
00612                 {
00613                     // 
00614                     // If we are here in the first place (i.e. after @getNymbox just removed
00615                     // all the messages in this sent buffer that already had a reply sitting
00616                     // in the nymbox) therefore we KNOW any messages in here never got a reply
00617                     // from the server, 
00618                     
00619                     const bool bReplyWasSuccess        = false; // If the msg had been a success, the reply (whether transaction within succeeded or failed) would have been dropped into my Nymbox, and thus removed from this "sent buffer" in @getNymbox.
00620                     const bool bReplyWasFailure        = true; // If the msg had been an explicit failure, the reply (without the transaction inside of it even having a chance to succeed or fail) would definitely NOT have been dropped into my Nymbox, and thus removed from this "sent buffer" in @getNymbox. However, IN THIS ONE CASE, since we DID just download the Nymbox and verify there ARE NO REPLIES for this request number (before calling this function), and since a dropped message is basically identical to a rejected message, since in either case, the transaction itself never even had a chance to run, we are able to now harvest the message AS IF the server HAD explicitly rejected the message. This is why I pass true here, where anywhere else in the code I would always pass false unless I had explicitly received a failure from the server. This place in the code, where we are now, is the failsafe endpoint for missed/dropped messages! IF they STILL haven't been found by this point, they are cleaned up as if the message was explicitly rejected by the server before the transaction even had a chance to run.
00621 
00622                     const bool bTransactionWasSuccess  = false; // Per above, since "the transaction never had a chance to run" then it could NOT have been an explicit success.
00623                     const bool bTransactionWasFailure  = false; // Per above, since "the transaction never had a chance to run" then it could NOT have been an explicit failure.
00624                     // -----------------------------------------------------
00625                     pThisMsg->HarvestTransactionNumbers(*pNym,      // Actually it's pNym who is "harvesting" the numbers in this call.   <========= HARVEST
00626                                                         *pbHarvestingForRetry,
00627                                                         bReplyWasSuccess,
00628                                                         bReplyWasFailure,
00629                                                         bTransactionWasSuccess,

00631                 } // if there's a transaction to be harvested inside this message.
00632             } // if pNym !NULL
00633             // ----------------------

00635             ++temp_it;

00638             // ----------------------
00639             delete pThisMsg;                // <============ DELETE
00640             pThisMsg = NULL;
00641             // ---------------------------------------------------------------------------
00642             if (NULL != pstrNymID && NULL != pstrServerID)
00643             {
00644                 OTString strFolder, strFile;

00646                                  OTFolders::Nym().Get(),     OTLog::PathSeparator(),

00648                                  "sent", /*todo hardcoding*/ OTLog::PathSeparator(),
00649                                  pstrNymID->Get());

00651                 // ---------------------------------------------------------------------------

00654                 if (OTDB::Exists(strFolder.Get(), str_data_filename))
00655                 {
00656                     OTString strNumList(OTDB::QueryPlainString(strFolder.Get(), str_data_filename));
00657                     if (strNumList.Exists())
00658                         theNumList.Add(strNumList);
00659                     theNumList.Remove(lRequestNum); // Clear (this function) loops and removes them. (Here's the one being removed this iteration.)
00660                 }
00661                 else // it doesn't exist on disk, so let's just create it from the list we have in RAM so we can store it to disk.
00662                 {    // NOTE: this may be unnecessary since we are "clear"ing them all anyway. But that just means we can remove this
00663                      // block during optimization. Todo optimize.
00664                      // Since we create the NumList based on m_mapMessages, and since the message for this iteration was already removed
00665                      // above, we don't need to remove anything at this point, we just create the NumList to contain the same numbers as are
00666                      // in m_mapMessages.
00667                      //
00668                     it = m_mapMessages.begin();
00669                     while (it != m_mapMessages.end())
00670                     {
00671                         // -----------------------------
00672                         const int64_t  & lTempReqNum   = it->first;
00673                         // -----------------------
00674                         OTMessage   * pMsg          = it->second;
00675                         OT_ASSERT(NULL != pMsg);
00676                         // -----------------------------
00677                         //
00678                         // If a server ID was passed in, but doesn't match the server ID on this message,
00679                         // Then skip this one. (Same with the NymID.)
00680                         //
00681                         if (!pstrServerID->Compare(pMsg->m_strServerID) ||
00682                             !pstrNymID->   Compare(pMsg->m_strNymID))
00683                         {
00684                             ++it;
00685                             continue;
00686                         }
00687                         // --------
00688                         else
00689                         {
00690                             theNumList.Add(lTempReqNum);
00691                         }
00692                         ++it;
00693                     }
00694                 }// else
00695                 // ----------------------------------
00696                 // By this point, theNumList has either been loaded from local storage and had the number removed,
00697                 // or it wasn't in local storage and thus we created it and added all the numbers to it from RAM (not
00698                 // including the one being erased, since it was already removed from the RAM list, above.) So either
00699                 // way, the number being removed is now ABSENT from theNumList.
00700                 //
00701                 // Therefore nothing left to do here, but save it back again!
00702                 //
00703                 OTString strOutput;
00704                 theNumList.Output(strOutput);
00705                 if (!OTDB::StorePlainString(strOutput.Get(), strFolder.Get(), str_data_filename)) // todo hardcoding.
00706                 {
00707                     OTLog::Error("OTMessageOutbuffer::Clear: Error: failed writing list of request numbers to storage.\n");
00708                 }
00709                 // ----------------------------------
00710                 // Make sure any messages being erased here, are also erased from local storage.
00711                 // Now that we've updated the numlist in local storage, let's
00712                 // erase the sent message itself...
00713                 //
00714                 OTMessage * pMsg = new OTMessage;
00715                 OT_ASSERT(NULL != pMsg);
00716                 OTCleanup<OTMessage> theMsgAngel(pMsg);
00717                 
00718                 if (OTDB::Exists(strFolder.Get(), strFile.Get()) && pMsg->LoadContract(strFolder.Get(), strFile.Get()))
00719                 {
00720                     OTDB::EraseValueByKey(strFolder.Get(), strFile.Get());
00721                 }
00722             }
00723             // ---------------------------------------------------------------------------
00724         }
00725     }
