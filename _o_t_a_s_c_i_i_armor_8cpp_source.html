00301 std::string compress_string(const std::string& str,
00302                             int32_t compressionlevel = Z_BEST_COMPRESSION)
00303 {
00304     z_stream zs;                        // z_stream is zlib's control structure
00305     memset(&zs, 0, sizeof(zs));
00306     
00307     if (deflateInit(&zs, compressionlevel) != Z_OK)
00308         throw(std::runtime_error("deflateInit failed while compressing."));
00309     
00310     zs.next_in = (Bytef*)str.data();
00311     zs.avail_in = static_cast<uInt>(str.size());           // set the z_stream's input
00312     
00313     int32_t ret;
00314     char outbuffer[32768];
00315     std::string outstring;
00316     
00317     // retrieve the compressed bytes blockwise
00318     do {
00319         zs.next_out = reinterpret_cast<Bytef*>(outbuffer);
00320         zs.avail_out = sizeof(outbuffer);
00321         
00322         ret = deflate(&zs, Z_FINISH);
00323         
00324         if (outstring.size() < zs.total_out) {
00325             // append the block to the output string
00326             outstring.append(outbuffer,
00327                              zs.total_out - outstring.size());
00328         }
00329     } while (ret == Z_OK);
00330     
00331     deflateEnd(&zs);
00332     
00333     if (ret != Z_STREAM_END) {          // an error occurred that was not EOF
00334         std::ostringstream oss;
00335         oss << "Exception during zlib compression: (" << ret << ") " << zs.msg;
00336         throw(std::runtime_error(oss.str()));
00337     }
00338     
00339     return outstring;
00340 }

00970 // This code reads up the string, discards the bookends, and saves only the gibberish itself.
00971 // the bEscaped option allows you to load a normal ASCII-Armored file if off, and allows
00972 // you to load an escaped ASCII-armored file (such as inside the contracts when the public keys
00973 // are escaped with a "- " before the rest of the ------- starts.)
00974 //
00975 bool OTASCIIArmor::LoadFromString(OTString & theStr, // input
00976                                   bool bEscaped/*=false*/, 
00977                                   const // This szOverride sub-string determines where the content starts, when loading.
00978                                   std::string str_override/*="-----BEGIN"*/) // Default is "-----BEGIN"
00979 {
00980     // Should never be 0 size, as default is "-----BEGIN"
00981     // But if you want to load a private key, try "-----BEGIN ENCRYPTED PRIVATE" instead.
00982     // *smile*
00983     const std::string str_end_line = "-----END"; // Someday maybe allow parameterized option for this.
00984     // ------------------------------------------
00985     const int32_t nBufSize  = 2100; // todo: hardcoding
00986     const int32_t nBufSize2 = 2048; // todo: hardcoding
00987     // -----------------------------
00988     char buffer1[2100];         // todo: hardcoding
00989     
00990     std::fill(&buffer1[0], &buffer1[(nBufSize-1)], 0); // Initializing to 0.
00991     
00992     bool bContentMode            = false;  // "Currently IN content mode."
00993     bool bHaveEnteredContentMode = false;  // "Have NOT YET entered content mode."
00994     
00995     // Clear out whatever string might have been in there before.
00996     Release();
00997     
00998     // Load up the string from theStr, 
00999     // (bookended by "-----BEGIN ... -----" and "END-----" messages)
01000     bool bIsEOF = false;
01001     theStr.reset(); // So we can call theStr.sgets(). Making sure position is at start of string.
01002     
01003     do
01004     {
01005         bIsEOF = !(theStr.sgets(buffer1, nBufSize2)); // 2048
01006         
01007         std::string line        = buffer1;  
01008         const char * pConstBuf  = line.c_str();
01009         char * pBuf             = (char *)pConstBuf;
01010         
01011         // It's not a blank line.
01012         if (line.length() < 2)
01013         {
01014             continue;
01015         }
01016         
01017         // if we're on a dashed line...
01018         else if (line.at(0) == '-' && 
01019                  line.at(2) == '-' && 
01020                  line.at(3) == '-' &&
01021                  (bEscaped ? (line.at(1) == ' ') : (line.at(1) == '-') )
01022                 )
01023         {           
01024             // If I just hit a dash, that means there are only two options:
01025             
01026             // a. I have not yet entered content mode, and potentially just now entering it for the first time.
01027             if (!bHaveEnteredContentMode)
01028             {
01029                 // str_override defaults to:  "-----BEGIN" (If you want to load a private key instead,
01030                 // Try passing "-----BEGIN ENCRYPTED PRIVATE" instead of going with the default.)
01031                 //
01032                 if (line.find(str_override) != std::string::npos && 
01033                     line.at(0) == '-' && 
01034                     line.at(2) == '-' && 
01035                     line.at(3) == '-' && 
01036                     (bEscaped ? (line.at(1) == ' ') : (line.at(1) == '-'))
01037                    )

01039 //                  OTLog::Error("Reading ascii-armored contents...");
01040                     bHaveEnteredContentMode = true;
01041                     bContentMode = true;
01042                     continue;
01043                 }
01044                 else
01045                 {
01046                     continue;
01047                 }               
01048             }
01049             
01050             // b. I am now LEAVING content mode!
01051             else if (bContentMode &&
01052                      // str_end_line is "-----END"
01053                      (line.find(str_end_line) != std::string::npos))
01054             {
01055 //              OTLog::Error("Finished reading ascii-armored contents.\n");
01056 //              OTLog::vError("Finished reading ascii-armored contents:\n%s(END DATA)\n", Get());
01057                 bContentMode   = false;
01058                 continue;
01059             }
01060         }
01061         
01062         // Else we're on a normal line, not a dashed line.
01063         else
01064         {
01065             if (bHaveEnteredContentMode && bContentMode)
01066             {

01076                 }
01077             }
01078             
01079         }

01082         if (bContentMode)
