00363 void * OTPassword::safe_memcpy(void   * dest,
00364                                uint32_t dest_size,
00365                                const
00366                                void   * src,
00367                                uint32_t src_length,
00368                                bool     bZeroSource/*=false*/) // if true, sets the source buffer to zero after copying is done.
00369 {
00370     bool bSuccess = false;
00371 
00372     // Make sure they aren't null.
00373     OT_ASSERT(NULL != dest);
00374     OT_ASSERT(NULL != src);
00375 
00376     // Make sure they aren't the same pointer.
00377     OT_ASSERT(src != dest);
00378 
00379     // Make sure it will fit.
00380     OT_ASSERT_MSG(src_length <= dest_size, "ASSERT: safe_memcpy: destination buffer too small.\n");
00381 
00382     // Make sure they don't overlap.
00383     // First assert does the beginning of the string, makes sure it's not within the bounds of the destination
00384     // string. Second assert does the same thing for the end of the string. Finally a third is needed to make sure
00385     // we're not in a situation where the beginning is less than the dest beginning, yet the end is also more than
00386     // the dest ending!
00387     //
00388     OT_ASSERT_MSG(false == ((static_cast<const uint8_t *>(src) >   static_cast<uint8_t *>(dest)) &&
00389                             (static_cast<const uint8_t *>(src) <  (static_cast<uint8_t *>(dest) + dest_size))),
00390                   "ASSERT: safe_memcpy: Unexpected memory overlap, start of src.\n");
00391     OT_ASSERT_MSG(false == (((static_cast<const uint8_t *>(src) + src_length) >   static_cast<uint8_t *>(dest)) &&
00392                             ((static_cast<const uint8_t *>(src) + src_length) <  (static_cast<uint8_t *>(dest) + dest_size))),
00393                   "ASSERT: safe_memcpy: Unexpected memory overlap, end of src.\n");
00394     OT_ASSERT(false ==  ((static_cast<const uint8_t *>(src) <= static_cast<uint8_t *>(dest)) &&
00395                         ((static_cast<const uint8_t *>(src) + src_length) >= (static_cast<uint8_t *>(dest) + dest_size))));
00396 
00397 #ifdef _WIN32
00398     bSuccess = (0 == memcpy_s(dest,         // destination
00399                               static_cast<size_t>(dest_size),    // size of destination buffer.
00400                               src,          // source
00401                               static_cast<size_t>(src_length))); // length of source.
00402 #else
00403     bSuccess = (memcpy(dest, src, static_cast<size_t>(src_length)) == dest);
00404 #endif
00405 
00406     if (bSuccess)

00410             OTPassword::zeroMemory(const_cast<void *>(src), src_length);

00762 // OTPassword::SetSize   (Low-level)
00763 //
00764 // There are certain weird cases, like in OTSymmetricKey::GetPassphraseFromUser,
00765 // where we set the password using the getPassword_writable, and it's properly
00766 // null-terminated, yet this instance still doesn't know its actual size (even though
00767 // the size is known.) Therefore I added this call in order to set the size in
00768 // those odd cases where it's necessary. That being said, YOU should normally NEVER
00769 // need to use this function, so just pretend it doesn't exist.
00770 //
00771 // This adds a null terminator, IF we're in text mode (not binary mode.)
00772 //
00773 bool OTPassword::SetSize(uint32_t uSize)
00774 {
00775     if (m_bIsBinary)
00776     {
00777         if (uSize > getBlockSize())
00778             uSize = getBlockSize(); // Truncated password beyond max size.
00779         m_nPasswordSize = uSize;
00780         return true;
00781     }
00782     else if (m_bIsText)
00783     {
00784         if (uSize >= getBlockSize())  // Cannot be as much as the blocksize, because no room for null-terminator.
00785             uSize = getBlockSize()-1; // Truncated password to blocksize-1.
00786         m_szPassword[uSize] = '\0';   // The actual null-terminator.
00787         m_nPasswordSize     = uSize;  // If size is 3, the terminator is at m_szPassword[3] (which is the 4th byte.)
00788         return true;
00789     }
00790     OTLog::vError("%s: Error: m_bIsBinary and m_bIsText are both false. (Should never happen.)\n", __FUNCTION__);
00791     return false;
00792 }

00954 // (FYI, truncates if nAppendSize + getPasswordSize() is larger than getBlockSize.)
00955 // Returns number of bytes appended, or -1 for error.
00956 //
00957 int32_t OTPassword::addMemory(const void * vAppend, uint32_t nAppendSize)
00958 {
00959     OT_ASSERT(NULL != vAppend);
00960 
00961 //  const char * szFunc = "OTPassword::addMemory";

00963     if (0 == nAppendSize)
00964         return 0;
00965     // ---------------------------------
00966     // If I'm currently at a 0 size, then call setMemory instead.
00967     //
00968     if (m_nPasswordSize == 0)
00969         return this->setMemory(vAppend, nAppendSize);
00970     // ***********************************************
00971     //
00972     // By this point, I know I already have some memory allocated,
00973     // and I'm actually appending some other memory onto the end of it.
00974     //

00976     // ---------------------------------
00977     // Make sure total new size isn't larger than our block size
00978     //
00979     if ((nAppendSize + m_nPasswordSize) > getBlockSize())
00980         nAppendSize = (getBlockSize() - m_nPasswordSize); // Truncated password beyond max size.
00981     // ---------------------------------
00982 //  OT_ASSERT(nAppendSize >= 0);
00983 
00984     if (0 == nAppendSize)
00985         return 0;

00987     // By this point, I know nAppendSize is larger than 0, AND that appending it onto the
00988     // existing memory of this object will not exceed the total allowed block size.
00989     //
00990     // Because we use setMemory when empty, and only use addMemory when we KNOW something
00991     // is already there, therefore we know the page is already locked, so no need to go
00992     // trying to lock it again.
00993     // ---------------------------------
00994     OTPassword::safe_memcpy(static_cast<void *>(&(m_szPassword[m_nPasswordSize])),
00995                             static_cast<uint32_t>(nAppendSize), // dest size is based on the source size, but guaranteed to be >0 and <=getBlockSize
00996                             vAppend,
00997                             static_cast<uint32_t>(nAppendSize)); // Since dest size is known to be src size or less (and >0) we use it as src size. (We may have truncated... and we certainly don't want to copy beyond our own truncation.)
