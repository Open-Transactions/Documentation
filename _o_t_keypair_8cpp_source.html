00641 int32_t OTKeypair::GetPublicKeyBySignature(listOfAsymmetricKeys & listOutput, // Inclusive means, return the key even when theSignature has no metadata.
00642                                        const OTSignature & theSignature, bool bInclusive/*=false*/) const
00643 {
00644     OT_ASSERT(NULL != m_pkeyPublic);
00645     OT_ASSERT(NULL != m_pkeyPublic->m_pMetadata);
00646     
00647     // We know that EITHER exact metadata matches must occur, and the signature MUST have metadata, (bInclusive=false)
00648     // OR if bInclusive=true, we know that metadata is still used to eliminate keys where possible, but that otherwise,
00649     // if the signature has no metadata, then the key is still returned, "just in case."
00650     //
00651     if ((false == bInclusive) && (false == theSignature.m_metadata.HasMetadata()))
00652         return 0;
00653     // -------------------------
00654     // Below this point, metadata is used if it's available.
00655     // It's assumed to be "okay" if it's not available, since any non-inclusive
00656     // calls would have already returned by now, if that were the case.
00657     // (But if it IS available, then it must match, or the key won't be returned.)
00658     //
00659     if ((false == theSignature.m_metadata.HasMetadata())    || // If the signature has no metadata,
00660         (false == m_pkeyPublic->m_pMetadata->HasMetadata()) || // Or if m_pkeyPublic has no metadata,
00661         (m_pkeyPublic->m_pMetadata->HasMetadata() &&           // OR if they BOTH have metadata, and
00662          theSignature.m_metadata.   HasMetadata() &&           // their metadata is a MATCH...
00663          (theSignature.m_metadata == *(m_pkeyPublic->m_pMetadata)) ) )
00664     {
00665         // ...Then add m_pkeyPublic as a possible match, to listOutput.
00666         //
00667         listOutput.push_back(m_pkeyPublic);
00668         return 1;
00669     }
00670     return 0;
00671 }

00684     // --------------------------------------
00685     // If we were importing, we were in the exported format but now we're in the internal format.
00686     // Therefore we want to use the wallet's internal cached master passphrase to save. Therefore
00687     // strReason will be used for the import case.
00688     //
00689     // But if we were exporting, then we were in the internal format and just re-encrypted to the

00691     //

00693                                   "Enter the new export passphrase. (Above ReEncryptPrivateKey in OTKeypair::ReEncrypt)" :

00698                                   "Enter the new export passphrase. (Below ReEncryptPrivateKey in OTKeypair::ReEncrypt)");
00699     // --------------------------------------
00700     // At this point the public key was loaded from a public key, not a cert,
00701     // but the private key was loaded from the cert. Therefore we'll save the
00702     // public cert from the private key, and then use that to reload the public
00703     // key after ReEncrypting. (Otherwise the public key would be there, but it
00704     // would be missing the x509, which is only available in the cert, not the

00708     // sticking the public key in there? Because not all key credentials will use

00710     // certs. But I might still just stick it in there, and code things to be able to
00711     // load either indiscriminately. After all, that's what I'm doing already in the
00712     // asset and server contracts. But even in those cases, there will be times when
00713     // only a pubkey is available, not a cert, so I'll probably still find myself having
00714     // to do this. Hmm...
00715     
00716     // ---------------------------------------
00717     
00718     const bool bReEncrypted = m_pkeyPrivate->ReEncryptPrivateKey(theExportPassword, bImporting); // <==== IMPORT or EXPORT occurs here.
00719           bool bGotCert = false;
00720     
00721     if (bReEncrypted)
00722     {
00723         // ------------------------------------------------
00724         // Keys won't be right until this happens. Todo: eliminate this need.
00725         bGotCert = this->SaveAndReloadBothKeysFromTempFile(&strOutput, &strReasonBelow,
00726                                                            bImporting ? NULL : &theExportPassword);
00727     }
00728     
00729     const bool bSuccess = (bReEncrypted && bGotCert);
00730     // --------------------------------------
