01092 bool OTTrade::ProcessCron()
01093 {
01094     // -----------------------------------------------------------------
01095     // Right now Cron is called 10 times per second.
01096     // I'm going to slow down all trades so they are once every GetProcessInterval()
01097     if (GetLastProcessDate() > OT_TIME_ZERO)
01098     {
01099         // (Default ProcessInterval is 1 second, but Trades will use 10 seconds,
01100         // and Payment Plans will use an hour or day.)
01101         if (OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetLastProcessDate()) <= GetProcessInterval())
01102             return true;
01103     }
01104     // ----------------------------------------------------------------
01105     // Keep a record of the last time this was processed.
01106     // (NOT saved to storage, only used while the software is running.)
01107     // (Thus no need to release signatures, sign contract, save contract, etc.)
01108     SetLastProcessDate(OTTimeGetCurrentTime());
01109     // -----------------------------------------------------------------
01110     // PAST END DATE?
01111     // First call the parent's version (which this overrides) so it has
01112     // a chance to check its stuff. Currently it checks IsExpired().
01113     if (false == ot_super::ProcessCron())

01115     // ----------------------------------------------------------------
01116     // You might ask, why not check here if this trade is flagged for removal?
01117     // Supposedly the answer is, because it's only below that I have the market pointer,
01118     // and am able to remove the corresponding trade from the market.
01119     // Therefore I am adding a hook for "onRemoval" so that Objects such as OTTrade ALWAYS
01120     // have the opportunity to perform such cleanup, without having to juggle such logic.
01121     // ----------------------------------------------------------------
01122     // REACHED START DATE?
01123     // Okay, so it's not expired. But might not have reached START DATE yet...
01124     if (!VerifyCurrentDate())

01126                         //  true, so it will stay on Cron until it BECOMES valid.
01127     // ----------------------------------------------------------------
01128     // TRADE-specific stuff below.
01129 
01130     bool bStayOnMarket = true; // by default stay on the market (until some rule expires me.)
01131 
01132     OTIdentifier OFFER_MARKET_ID;
01133     OTMarket * pMarket = NULL;
01134     // ----------------------------------------------------------------
01135     // If the Offer is already active on a market, then I already have a pointer to
01136     // it. This function returns that pointer. If NULL, it tries to find the offer on
01137     // the market and then sets the pointer and returns. If it can't find it, IT TRIES
01138     // TO ADD IT TO THE MARKET and sets the pointer and returns it.
01139     OTOffer * pOffer = GetOffer(&OFFER_MARKET_ID, &pMarket); // Both of these parameters are optional.
01140     // ----------------------------------------------------------------
01141     // In this case, the offer is NOT on the market.
01142     // Perhaps it wasn't ready to activate yet.
01143     if (NULL == pOffer)

01154     // ----------------------------------------------------------------

01158 //      OTLog::Error("OTTrade::ProcessCron: Market was NULL.\n"); // comment this out
01159     }

01161     else  // If a valid pointer was returned, that means the offer is on the market.
01162     {
01163         // Make sure it hasn't already been flagged by someone else...
01164         if (this->IsFlaggedForRemoval())    // This is checked above in OTCronItem::ProcessCron().
01165             bStayOnMarket = false;          // I'm leaving the check here in case the flag was set since then.
01166         // ---------------------------------------
01167         else // Process it!  <===================
01168         {
01169             OTLog::vOutput(2, "Processing trade: %lld.\n", GetTransactionNum());
