00147 bool isPowerOfTen( const int64_t & x )
00148 {
00149     if (1 == x)
00150         return true;
00151     
00152     const int64_t lBase = 10;
00153     int64_t lIt = lBase;
00154     
00155     for (int32_t i = 1; i < 23; i++)
00156     {
00157         if (x == lIt)
00158             return true;
00159         lIt *= lBase;
00160     }
00161 
00162     return false;
00163 }

00166 /*
00167  Let's say you wanted to add an Offer to a Market. But you don't know
00168  which market.  There are different markets for different combinations
00169  of asset and currency. There are also higher and lower level markets
00170  for different trade minimums.  
00171  
00172  The server has to be able to match up your Offer to the Right Market,
00173  so that it can trade with similar offers.
00174  
00175  So in this method, I combine the Asset Type ID, the Currency Type ID,
00176  and the minimum increment, and use them to generate a UNIQUE ID, which
00177  will also be the same, given the same input.
00178  
00179  That is the ID I will use for looking up the offers on the market.
00180  Basically it's the Market ID, and the Offer just has the SAME ID,
00181  and that's how you match it up to the market.
00182  
00183  (This is analogous to how Transactions and Transaction Items have the
00184  same transaction number.)
00185  
00186  THIS MEANS that the user cannot simply set his minimum increment to
00187  a "divide-into equally" with the market minimum increment. Why not?
00188  Because since his number will be different from the next guy, they
00189  will calculate different IDs and thus end up on different markets.
00190 
00191  TODO: therefore the user MUST SUPPLY the EXACT SAME minimum increment
00192  of the market he wishes to trade on. There's no other way. However,
00193  I CAN allow the user to ALSO provide a second minimum, which must be
00194  a multiple of the first.
00195  
00196  TODO: Should add this same method to the Market object as well.
00197  
00198  
00199  To use OTOffer::GetIdentifier is simple:
00200  
00201  void blah (OTOffer & theOffer)
00202  {
00203     OTIdentifier MARKET_ID(theOffer); // the magic happens right here.
00204  
00205     // (Done.)
00206  }
00207  */

00427 bool OTOffer::MakeOffer(bool   bBuyingOrSelling,    // True == SELLING, False == BUYING
00428                   const int64_t & lPriceLimit,         // Per Minimum Increment... (Zero price means it's a market order.)
00429                   const int64_t & lTotalAssetsOffer,   // Total assets available for sale or purchase.
00430                   const int64_t & lMinimumIncrement,   // The minimum increment that must be bought or sold for each transaction
00431                   const int64_t & lTransactionNum,     // The transaction number authorizing this trade.
00432                   const time64_t & VALID_FROM/*=0*/,  // defaults to RIGHT NOW
00433                   const time64_t & VALID_TO/*=0*/)    // defaults to 24 hours (a "Day Order")
00434 {
00435     m_bSelling              = bBuyingOrSelling;     // Bid or Ask?
00436     SetTransactionNum       (lTransactionNum);
00437     SetTotalAssetsOnOffer   (lTotalAssetsOffer);    // 500 bushels for sale.
00438     
00439     m_strContractType.Set((m_bSelling ? "ASK" : "BID"));
00440 
00441     // Make sure minimum increment isn't bigger than total Assets.
00442     // (If you pass them into this function as the same value, it's functionally a "FILL OR KILL" order.)
00443     int64_t lRealMinInc = lMinimumIncrement;
00444     if (lMinimumIncrement > lTotalAssetsOffer)      // Once the total, minus finish so far, is smaller than the minimum increment,
00445         lRealMinInc = lTotalAssetsOffer;            // then the OTTrade object I am linked to will expire and remove me from the market.
00446                                                     // OR it could set the minimum increment to the remainder. But then need to calc price.
00447     
00448     SetMinimumIncrement     (lRealMinInc);          // Must sell in 50 bushel increments. (Perhaps on the 10-bushel market it will sell in 5 increments of 10.)
00449     SetPriceLimit           (lPriceLimit);          // Won't sell for any less than $10 per increment. (Always get best market price.)
00450     SetFinishedSoFar        (0);                    // So far have already sold 350 bushels. Actual amount available is (total - finished).
00451     
00452     time64_t REAL_VALID_FROM    = VALID_FROM;
00453     time64_t REAL_VALID_TO  = VALID_TO;
00454     
00455     if (OT_TIME_ZERO >= VALID_FROM)
00456     {
00457         REAL_VALID_FROM = OTTimeGetCurrentTime(); // This time is set to TODAY NOW
00458     }
00459     
00460     if (OT_TIME_ZERO >= VALID_TO)
00461     {
00462         // (All offers default to a "DAY ORDER" if valid dates not specified.)
00463         REAL_VALID_TO = OTTimeAddTimeInterval(REAL_VALID_FROM, OTTimeGetSecondsFromTime(OT_TIME_DAY_IN_SECONDS)); // 1 day. 
00464     }
00465     
00466     SetValidFrom(REAL_VALID_FROM);
00467     SetValidTo(REAL_VALID_TO);
00468 
00469     return true;
00470 }
