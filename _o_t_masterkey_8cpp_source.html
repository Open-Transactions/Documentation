00280 // Should actually curl the URL, or lookup the blockchain value, or verify Cert against
00281 // Cert Authority, etc. Due to the network slowdown of this step, we will eventually make
00282 // a separate identity verification server.
00283 //
00284 bool OTMasterkey::VerifyAgainstSource() const
00285 {
00286     // RULE: *Any* source except for a public key, will begin with a
00287     // protocol specifier. Such as:
00288     //
00289     // http:        (a normal URL)
00290     // https:       (a normal URL on https)
00291     // bitcoin:     (a bitcoin address)
00292     // namecoin:    (a namecoin address)
00293     // i2p:         (an i2p address)
00294     // tor:         (a tor address)
00295     // freenet:     (a freenet address)
00296     // cert:        (Subject and Issuer DN from the cert)
00297     //
00298     // If NO protocol specifier is found, the source is assumed
00299     // to be a public key.
00300     // Public key is the default because that's the original behavior
00301     // of OT anyway: the public key was hashed to form the NymID. We will
00302     // continue to support this as a default, but now we are additionally
00303     // also allowing other sources such as Namecoin, Freenet, etc. As int64_t
00304     // as a Nym's source hashes to its correct ID, and as long as its master
00305     // credentials can be verified from that same source, then all master
00306     // credentials can be verified (as well as subcredentials) from any source
00307     // the user prefers.
00308     //
00309     // ---------------------------------
00310     bool bVerified = false;
00311     // ---------------------------------
00312     const std::string str_raw_source(m_strSourceForNymID.Get());
00313     std::string str_source;
00314     
00315     // It's a URL.
00316     if (str_raw_source.compare(0,5,"http:")  == 0)
00317     {
00318         str_source.insert(str_source.begin(), str_raw_source.begin()+5, str_raw_source.end());
00319         bVerified = this->VerifySource_HTTP(str_source.c_str());
00320     }
00321     else if (str_raw_source.compare(0,6,"https:")  == 0)
00322     {
00323         str_source.insert(str_source.begin(), str_raw_source.begin()+6, str_raw_source.end());
00324         bVerified = this->VerifySource_HTTPS(str_source.c_str());
00325     }
00326     // It's a Bitcoin address.
00327     else if (str_raw_source.compare(0,8,"bitcoin:") == 0)
00328     {
00329         str_source.insert(str_source.begin(), str_raw_source.begin()+8, str_raw_source.end());
00330         bVerified = this->VerifySource_Bitcoin(str_source.c_str());
00331     }
00332     // It's a Namecoin address.
00333     else if (str_raw_source.compare(0,9,"namecoin:") == 0)
00334     {
00335         str_source.insert(str_source.begin(), str_raw_source.begin()+9, str_raw_source.end());
00336         bVerified = this->VerifySource_Namecoin(str_source.c_str());
00337     }
00338     // It's a Freenet URL.
00339     else if (str_raw_source.compare(0,8,"freenet:") == 0)
00340     {
00341         str_source.insert(str_source.begin(), str_raw_source.begin()+8, str_raw_source.end());
00342         bVerified = this->VerifySource_Freenet(str_source.c_str());
00343     }
00344     // It's a Tor URL.
00345     else if (str_raw_source.compare(0,4,"tor:") == 0)
00346     {
00347         str_source.insert(str_source.begin(), str_raw_source.begin()+4, str_raw_source.end());
00348         bVerified = this->VerifySource_TOR(str_source.c_str());
00349     }
00350     // It's an I2P URL.
00351     else if (str_raw_source.compare(0,4,"i2p:") == 0)
00352     {
00353         str_source.insert(str_source.begin(), str_raw_source.begin()+4, str_raw_source.end());
00354         bVerified = this->VerifySource_I2P(str_source.c_str());
00355     }
00356     // It's the Issuer/Subject DN info from a cert issued by a traditional certificate authority.
00357     else if (str_raw_source.compare(0,5,"cert:") == 0)
00358     {
00359         str_source.insert(str_source.begin(), str_raw_source.begin()+5, str_raw_source.end());
00360         bVerified = this->VerifySource_CA(str_source.c_str());
00361     }
00362     else // It's presumably a public key.
00363     {
00364         str_source = str_raw_source;
00365         bVerified = this->VerifySource_Pubkey(str_source.c_str());
00366     }
00367     // -----------------------------------------------------------------
00368     return bVerified;
00369 }
