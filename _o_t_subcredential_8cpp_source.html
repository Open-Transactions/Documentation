00558                               __FUNCTION__, __FILE__, __LINE__, nCount, mapPrivate.size());
00559                 return (-1); // error condition
00560             }
00561             // --------------------------------
00562             OT_ASSERT(NULL != m_pOwner);
00563             
00564             // Sometimes we are supposed to use a specific, pre-specified password (versus just
00565             // blindly asking the user to type a password when it's not cached alrady.) For example,
00566             // when importing a Nym, the exported version of the Nym is not encrypted to the cached
00567             // wallet key. It's got its own exported passphrase. So it won't be cached, and we will have
00568             // the wallet's cached key disabled while loading it. This means we have to enter the same
00569             // passphrase many times in a row, while OT loads up all the credentials and keys for that
00570             // Nym. Therefore, instead, we ask the user up front to enter the special passphrase for that
00571             // exported Nym. Then we just pass it in to all the functions that need it, so none of them have
00572             // to ask the user to type it.
00573             //
00574             // That's what brings us here now... when that happens, m_pOwner->GetImportPassword() will be set
00575             // with the appropriate pointer to the passphrase. Otherwise it will be NULL. Meanwhile SetPrivateContents
00576             // below accepts an import passphrase, which it defaults to NULL.
00577             //
00578             // So we just pass it in either way (sometimes it's NULL and the wallet cached master key is used, and
00579             // sometimes an actual passphrase is passed in, so we use it.)
00580             
00581             if (false == this->SetPrivateContents(mapPrivate, m_pOwner->GetImportPassword()))
00582             {
00583                 OTLog::vError("%s, %s, %d: Subcredential failed setting private contents while loading.\n",
00584                               __FUNCTION__, __FILE__, __LINE__);
00585                 return (-1); // error condition    
00586             }
00587             // --------------------------------            
00588         } // if strCount.Exists() && nCount > 0
00589         // -------------------------------------------------------------------------------
00590         OTLog::Output(2, "Loaded privateContents for subcredential.\n");
00591         
00592         nReturnVal = 1;
00593     }
00594     // ----------------------------------
00595     return nReturnVal;
00596 }

00625 // Call VerifyNymID. Then verify m_strMasterCredID against the hash of
00626 // m_pOwner->GetMasterkey().GetPubCredential() (the master credential.) Verify that
00627 // m_pOwner->GetMasterkey() and *this have the same NymID. Then verify the
00628 // signature of m_pOwner->GetMasterkey().
00629 //
00630 bool OTSubcredential::VerifyInternally()
00631 {
00632     OT_ASSERT(NULL != m_pOwner);
00633     // ---------------------------------------
00634     // Verify that m_strNymID is the same as the hash of m_strSourceForNymID.
00635     //
00636     if (false == this->VerifyNymID())
00637         return false;
00638     // ---------------------------------------
00639     // Verify that m_pOwner->GetMasterkey() and *this have the same NymID.
00640     //
00641     if (false == m_strNymID.Compare(m_pOwner->GetMasterkey().GetNymID()))
00642     {
00643         OTLog::vOutput(0, "%s: Failure: The actual master credential's NymID doesn't match the NymID on this subcredential.\n"
00644                        "    This NymID: %s\nMaster's NymID: %s\n My Master Cred ID: %s\n", __FUNCTION__,
00645                        m_strNymID.Get(), m_pOwner->GetMasterkey().GetNymID().Get(),
00646                        m_strMasterCredID.Get());
00647         return false;
00648     }
00649     // ---------------------------------------
00650     // Verify m_strMasterCredID against the hash of m_pOwner->GetMasterkey().GetPubCredential()
00651     // (the master credentialID is a hash of the master credential.)
00652     //
00653     OTIdentifier theActualMasterID;
00654     const bool bCalcMasterCredID = theActualMasterID.CalculateDigest(m_pOwner->GetPubCredential());
00655     OT_ASSERT(bCalcMasterCredID);
00656     const OTString strActualMasterID(theActualMasterID);
00657  
00658     if (false == m_strMasterCredID.Compare(strActualMasterID))
00659     {
00660         OTLog::vOutput(0, "%s: Failure: When the actual Master Credential is hashed, the "
00661                        "result doesn't match the expected Master Credential ID.\n"
00662                        "Expected: %s\n   Found: %s\nMaster Cred:\n%s\n", __FUNCTION__,
00663                        m_strMasterCredID.Get(), strActualMasterID.Get(),
00664                        m_pOwner->GetPubCredential().Get());
00665         return false;
00666     }
00667     // ---------------------------------------
00668     // Then verify the signature of m_pOwner->GetMasterkey()...
00669     // Let's get a few things straight:
00670     // * OTMasterkey is a key (derived from OTKeyCredential, derived from OTSubcredential) and it can only sign itself.
00671     // * The only further verification a Masterkey can get is if its hash is posted at the source. Or, if the source
00672     //   is a public key, then the master key must be signed by the corresponding private key. (Again: itself.)
00673     // * Conversely to a master key which can ONLY sign itself, all subkeys must ALSO sign themselves.
00674     //
00675     // * Thus: Any OTKeyCredential (both master and subkeys, but no other credentials) must ** sign itself.**
00676     // * Whereas m_strMasterSigned is only used by OTSubkey, and thus only must be verified there.
00677     // * Any OTSubcredential must also be signed by its master. (Except masters, which already sign themselves.)
00678     // * Any OTMasterkey must (at some point, and/or regularly) verify against its own source.
00679     //
00680     // -----------------
00681     // * Any OTSubcredential must also be signed by its master. (Except masters, which already sign themselves.)
00682     //
00683     if (false == this->VerifySignedByMaster())
00684     {
00685         OTLog::vOutput(0, "%s: Failure: This subcredential hasn't been signed by its master credential.\n",
00686                        __FUNCTION__);
00687         return false;
00688     }
00689     // ---------------------------------------
00690     return true;
00691 }
