00166 // NOTE: You might ask, if we are using theSignature's metadata to narrow down the key type,
00167 // then why are we still passing the key type as a separate parameter? Good question. Because
00168 // often, theSignature will have no metadata at all! In that case, normally we would just NOT
00169 // return any keys, period. Because we assume, if a key credential signed it, then it WILL have
00170 // metadata, and if it doesn't have metadata, then a key credential did NOT sign it, and therefore
00171 // we know from the get-go that none of the keys from the key credentials will work to verify it,
00172 // either. That's why, normally, we don't return any keys if theSignature has no metadata.
00173 // BUT...Let's say you know this, that the signature has no metadata, yet you also still believe
00174 // it may be signed with one of these keys. Further, while you don't know exactly which key it
00175 // actually is, let's say you DO know by context that it's a signing key, or an authentication key,
00176 // or an encryption key. So you specify that. In which case, OT should return all possible matching
00177 // pubkeys based on that 1-letter criteria, instead of its normal behavior, which is to return all
00178 // possible matching pubkeys based on a full match of the metadata.
00179 //
00180 int32_t OTKeyCredential::GetPublicKeysBySignature(listOfAsymmetricKeys & listOutput,
00181                                               const OTSignature & theSignature,
00182                                               char cKeyType/*='0'*/) const // 'S' (signing key) or 'E' (encryption key) or 'A' (authentication key)
00183 {
00184     // Key type was not specified, because we only want keys that match the metadata on theSignature.
00185     // And if theSignature has no metadata, then we want to return 0 keys.
00186     if ((  '0' == cKeyType) &&
00187         (false == theSignature.m_metadata.HasMetadata()))
00188         return 0;
00189     // -----------------
00190     // By this point, we know that EITHER exact metadata matches must occur, and the signature DOES have metadata, ('0')
00191     // OR the search is only for 'A', 'E', or 'S' candidates, based on cKeyType, and that the signature's metadata
00192     // can additionally narrow the search down, if it's present, which in this case it's not guaranteed to be.
00193     int32_t nCount = 0;
00194     switch (cKeyType)
00195     {
00196             // Specific search only for signatures with metadata.
00197         case '0':
00198         {   // FYI, theSignature.m_metadata.HasMetadata() is true, in this case.
00199             // That's why I can just assume theSignature has a key type here:
00200             switch (theSignature.m_metadata.GetKeyType())
00201             {
00202                 case 'A': nCount = m_AuthentKey.GetPublicKeyBySignature(listOutput, theSignature); break; // bInclusive=false by default
00203                 case 'E': nCount = m_EncryptKey.GetPublicKeyBySignature(listOutput, theSignature); break; // bInclusive=false by default
00204                 case 'S': nCount = m_SigningKey.GetPublicKeyBySignature(listOutput, theSignature); break; // bInclusive=false by default
00205                 default: OTLog::vError("%s: Unexpected value for theSignature.m_metadata.GetKeyType: %c (failure)\n",
00206                                        __FUNCTION__, theSignature.m_metadata.GetKeyType()); return 0;
00207             }
00208             break;
00209         }
00210             // Generalized search which specifies key type and returns keys
00211             // even for signatures with no metadata. (When metadata is present,
00212             // it's still used to eliminate keys.)
00213         case 'A': nCount = m_AuthentKey.GetPublicKeyBySignature(listOutput, theSignature, true); break; // bInclusive=true
00214         case 'E': nCount = m_EncryptKey.GetPublicKeyBySignature(listOutput, theSignature, true); break; // bInclusive=true
00215         case 'S': nCount = m_SigningKey.GetPublicKeyBySignature(listOutput, theSignature, true); break; // bInclusive=true
00216         default:  OTLog::vError("%s: Unexpected value for cKeyType (should be 0, A, E, or S): %c\n",
00217                                 __FUNCTION__, cKeyType); return 0;
00218     }
00219     return nCount;
00220 }

00589     // If success, we now have the updated versions of the private certs.
00590     // 
00591     if (bSuccessReEncrypting)
00592     {
00593         mapOfStrings mapPrivate;        
00594         // ----------------------------------------
00595         FOR_EACH(mapOfStrings, m_mapPrivateInfo)
00596         {
00597             std::string  str_key_type     = (*it).first;   // A, E, S.
00598             std::string  str_key_contents = (*it).second;
00599             // ----------------------------------------
00600             if ("A" == str_key_type)

00602                 mapPrivate.insert(std::pair<std::string, std::string>("A", strAuth.Get()));
00603             }

00608             // ----------------------------------------

00610             {
00611                 mapPrivate.insert(std::pair<std::string, std::string>(str_key_type, str_key_contents));

00613             }

00621             // Logic: If I'm IMPORTING, bImporting is true, and that means the Nym WAS
00622             // encrypted to its own external passphrase (whenever it was exported) and
00623             // in order to IMPORT it, I re-encrypted all the keys from the exported passphrase,

00634             //
