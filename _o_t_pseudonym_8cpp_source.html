00211  the Nym and also to send him messages, verify his signatures on instruments, etc.
00212 
00213  pNym->LoadPublicKey() should load the UNSIGNED Nymfile from: ~/.ot/client_data/pubkeys/NYM_ID.cred
00214  Which, in the course of loading, should automatically load all its credentials from
00215  ~/.ot/client_data/pubkeys/credentials/NYM_ID/CREDENTIAL_ID
00216

00218  their signatures, and which verifies them against their source, which can also be hashed to

01103 // Whenever the number of transactions changes, you must sign the new list so you
01104 // aren't responsible for cleared transactions, for example. Or so you server will
01105 // allow you to take responsibility for a new transaction number (only if you've
01106 // signed off on it!)

01109 // a transaction (like a processNymbox!) Otherwise you would need a transaction number

01124     // since it uses up a transaction number, I will be sure to remove that one from my list before signing the list.

01769 // ** ResyncWithServer **
01770 //
01771 // Not for normal use! (Since you should never get out of sync with the server in the first place.)
01772 // However, in testing, or if some bug messes up some data, or whatever, and you absolutely need to
01773 // re-sync with a server, and you trust that server not to lie to you, then this function will do the trick.
01774 // NOTE: Before calling this, you need to do a getNymbox() to download the latest Nymbox, and you need to do
01775 // a createUserAccount() to download the server's copy of your Nym. You then need to load that Nymbox from
01776 // local storage, and you need to load the server's message Nym out of the @createUserAccount reply, so that
01777 // you can pass both of those objects into this function, which must assume that those pieces were already done
01778 // just prior to this call.
01779 //
01780 bool OTPseudonym::ResyncWithServer(OTLedger & theNymbox, OTPseudonym & theMessageNym)
01781 {
01782     bool bSuccess = true;
01783 
01784     // --------------------------------------
01785     const OTIdentifier &    theServerID = theNymbox.GetRealServerID();
01786     const OTString          strServerID(theServerID);
01787     const OTString          strNymID(m_nymID);
01788 
01789     const int32_t nIssuedNumCount   = theMessageNym.GetIssuedNumCount(theServerID);
01790     const int32_t nTransNumCount    = theMessageNym.GetTransactionNumCount(theServerID);
01791 
01792     // --------------------------------------
01793     // Remove all issued, transaction, and tentative numbers for a specific server ID,
01794     // as well as all acknowledgedNums, and the highest transaction number for that serverID,
01795     // from *this nym. Leave our record of the highest trans num received from that server,
01796     // since we will want to just keep it when re-syncing. (Server doesn't store that anyway.)
01797     //
01798     this->RemoveAllNumbers(&strServerID, false); // bRemoveHighestNum=true by default. But in this case, I keep it.
01799 
01800     // --------------------------------------
01801     // Any issued or trans numbers we add to *this from theMessageNym, are also added here so
01802     // they can be used to update the "highest number" record (at the bottom of this function.)
01803     //
01804     std::set<int64_t> setTransNumbers;
01805 
01806     // ***************************************************************************
01807     // Now that *this has no issued or transaction numbers for theServerID, we add
01808     // them back again from theMessageNym. (So they will match, and be 'N SYNC!!!)
01809     //
01810     // Copy the issued and transaction numbers from theMessageNym onto *this.
01811     //
01812     for(int32_t n1 = 0; n1 < nIssuedNumCount; ++n1)
01813     {
01814         const int64_t lNum = theMessageNym.GetIssuedNum(theServerID, n1);
01815 
01816         if (false == this->AddIssuedNum(strServerID, lNum))     // Add to list of numbers that haven't been closed yet.
01817         {
01818             OTLog::vError("OTPseudonym::ResyncWithServer: Failed trying to add IssuedNum (%lld) onto *this nym: %s, for server: %s\n",
01819                           lNum, strNymID.Get(), strServerID.Get());
01820             bSuccess = false;
01821         }
01822         else
01823         {
01824             setTransNumbers.insert(lNum);
01825             // ---------------------------
01826             OTLog::vOutput(1, "OTPseudonym::ResyncWithServer: Added IssuedNum (%lld) onto *this nym: %s, for server: %s \n",
01827                            lNum, strNymID.Get(), strServerID.Get());
01828         }
01829     }
01830     // --------------------------------------
01831     for(int32_t n2 = 0; n2 < nTransNumCount; ++n2)
01832     {
01833         const int64_t lNum = theMessageNym.GetTransactionNum(theServerID, n2);
01834 
01835         if (false == this->AddTransactionNum(strServerID, lNum))        // Add to list of available-to-use numbers.
01836         {
01837             OTLog::vError("OTPseudonym::ResyncWithServer: Failed trying to add TransactionNum (%lld) onto *this nym: %s, for server: %s\n",
01838                           lNum, strNymID.Get(), strServerID.Get());
01839             bSuccess = false;
01840         }
01841         else
01842         {
01843             setTransNumbers.insert(lNum);
01844             // ---------------------------
01845             OTLog::vOutput(1, "OTPseudonym::ResyncWithServer: Added TransactionNum (%lld) onto *this nym: %s, for server: %s \n",
01846                            lNum, strNymID.Get(), strServerID.Get());
01847         }
01848     }
01849     // ***************************************************************************
01850     // We already cleared all tentative numbers from *this (above in
01851     // RemoveAllNumbers). Next, loop through theNymbox and add Tentative numbers
01852     // to *this based on each successNotice in the Nymbox. This way, when the notices
01853     // are processed, they will succeed because the Nym will believe he was expecting them.
01854     //
01855     FOR_EACH(mapOfTransactions, theNymbox.GetTransactionMap())
01856     {
01857         OTTransaction * pTransaction = (*it).second;
01858         OT_ASSERT(NULL != pTransaction);
01859 //      OTString strTransaction(*pTransaction);
01860 //      OTLog::vError("TRANSACTION CONTENTS:\n%s\n", strTransaction.Get());
01861

01863         if ((OTTransaction::successNotice != pTransaction->GetType())) // if !successNotice
01864             continue;
01865         // -------------
01866         const int64_t lNum = pTransaction->GetReferenceToNum(); // successNotice is inRefTo the new transaction # that should be on my tentative list.
01867 
01868         if (false == this->AddTentativeNum(strServerID, lNum)) // Add to list of tentatively-being-added numbers.
01869         {
01870             OTLog::vError("OTPseudonym::ResyncWithServer: Failed trying to add TentativeNum (%lld) onto *this nym: %s, for server: %s\n",
01871                           lNum, strNymID.Get(), strServerID.Get());
01872             bSuccess = false;
01873         }
01874         else
01875             OTLog::vOutput(1, "OTPseudonym::ResyncWithServer: Added TentativeNum (%lld) onto *this nym: %s, for server: %s \n",
01876                            lNum, strNymID.Get(), strServerID.Get());
01877         // There's no "else insert to setTransNumbers" here, like the other two blocks above.
01878         // Why not? Because setTransNumbers is for updating the Highest Trans Num record on this Nym,
01879         // and the Tentative Numbers aren't figured into that record until AFTER they are accepted
01880         // from the Nymbox. So I leave them out, since this function is basically setting us up to
01881         // successfully process the Nymbox, which will then naturally update the highest num record
01882         // based on the tentatives, as it's removing them from the tentative list and adding them to
01883         // the "available" transaction list (and issued.)
01884 
01885     } // FOR_EACH (Nymbox)
01886     // ***************************************************************************
01887 
01888     const std::string strID = strServerID.Get();
01889 
01890     FOR_EACH_IT(mapOfHighestNums, this->m_mapHighTransNo, it_high_num)
01891     {
01892         if ( strID == it_high_num->first )  // We found it!
01893         {
01894             // -------------------------
01895             // See if any numbers on the set are higher, and if so, update the record to match.
01896             //
01897             FOR_EACH(std::set<int64_t>, setTransNumbers)
01898             {
01899                 const int64_t lTransNum = (*it);
01900                 // --------------------------------
01901                 // Grab a copy of the old highest trans number
01902                 const int64_t lOldHighestNumber = it_high_num->second;
01903                 // --------------------------------
01904                 if (lTransNum > lOldHighestNumber) // Did we find a bigger one?
01905                 {
01906                     // Then update the Nym's record!
01907                     this->m_mapHighTransNo[it_high_num->first] = lTransNum;
01908                     OTLog::vOutput(1, "OTPseudonym::ResyncWithServer: Updated HighestNum (%lld) record on *this nym: %s, for server: %s \n",
01909                                    lTransNum, strNymID.Get(), strServerID.Get());
01910                 }
01911             }
01912             // -------------------------
01913             // We only needed to do this for the one server, so we can break now.
01914             break;
01915         }
01916     }
01917     // ***************************************************************************
01918 
01919     return (this->SaveSignedNymfile(*this) && bSuccess);
01920 }

02480 void OTPseudonym::HarvestTransactionNumbers(const OTIdentifier & theServerID,
02481                                                   OTPseudonym  & SIGNER_NYM,
02482                                                   OTPseudonym  & theOtherNym, bool bSave/*=true*/)
02483 {
02484     const char * szFunc = "OTPseudonym::HarvestTransactionNumbers";
02485     // --------------------
02486     int64_t lTransactionNumber = 0;
02487 
02488     std::set<int64_t> setInput, setOutputGood, setOutputBad;
02489 
02490     FOR_EACH(mapOfTransNums, theOtherNym.GetMapIssuedNum())
02491     {
02492         std::string strServerID     = (*it).first;
02493         dequeOfTransNums * pDeque   = (it->second);
02494 
02495         OT_ASSERT(NULL != pDeque);
02496 
02497         OTString OTstrServerID = strServerID.c_str();
02498         const OTIdentifier theTempID(OTstrServerID);
02499 
02500         if (!(pDeque->empty()) && (theServerID == theTempID) ) // only for the matching serverID.
02501         {
02502             for (uint32_t i = 0; i < pDeque->size(); i++)
02503             {
02504                 lTransactionNumber = pDeque->at(i);
02505 
02506                 // If number wasn't already on issued list, then add to BOTH lists.
02507                 // Otherwise do nothing (it's already on the issued list, and no longer
02508                 // valid on the available list--thus shouldn't be re-added there anyway.)
02509                 //
02510                 if ((true   == VerifyTentativeNum(OTstrServerID, lTransactionNumber)) &&    // If I've actually requested this number and waiting on it...
02511                     (false  == VerifyIssuedNum(OTstrServerID, lTransactionNumber))          // and if it's not already on my issued list...
02512                     )
02513                     setInput.insert(lTransactionNumber);
02514             }
02515             break;  // We found it! Might as well break out.
02516         }
02517     } // for
02518 
02519     // ----------------------------------
02520     // Looks like we found some numbers to harvest
02521     // (tentative numbers we had already been waiting for,
02522     // yet hadn't processed onto our issued list yet...)
02523     //
02524     if (setInput.size() > 0)
02525     {
02526         const OTString strServerID(theServerID), strNymID(m_nymID);
02527 
02528         int64_t lViolator = this->UpdateHighestNum(SIGNER_NYM, strServerID, setInput,
02529                                                 setOutputGood, setOutputBad); // bSave=false (saved below already, if necessary)
02530 
02531         // NOTE: Due to the possibility that a server reply could be processed twice (due to redundancy
02532         // for the purposes of preventing syncing issues) then we expect we might get numbers in here
02533         // that are below our "last highest num" (due to processing the same numbers twice.) Therefore
02534         // we don't need to assume an error in this case. UpdateHighestNum() is already smart enough to
02535         // only update based on the good numbers, while ignoring the bad (i.e. already-processed) ones.
02536         // Thus we really only have a problem if we receive a (-1), which would mean an error occurred.
02537         // Also, the above call will log an FYI that it is skipping any numbers below the line, so no need
02538         // to log more in the case of lViolater being >0 but less than the 'last highest number.'
02539         //
02540         if ((-1) == lViolator)
02541             OTLog::vError("%s: ERROR: UpdateHighestNum() returned (-1), which is an error condition. "
02542                           "(Should never happen.)\nNym ID: %s \n", szFunc, strNymID.Get());
02543         else
02544         {
02545             // We only remove-tentative-num/add-transaction-num for the numbers that were above our 'last highest number'.
02546             // The contents of setOutputBad are thus ignored for these purposes.
02547             //
02548             FOR_EACH(std::set<int64_t>, setOutputGood)
02549             {
02550                 const int64_t lNoticeNum = (*it);
02551                 // --------------------------------------
02552                 // We already know it's on the TentativeNum list, since we checked that in the above for loop.
02553                 // We also already know that it's not on the issued list, since we checked that as well.
02554                 // That's why the below calls just ASSUME those things already.
02555                 //
02556                 RemoveTentativeNum(strServerID, lNoticeNum); // doesn't save (but saved below)
02557                 AddTransactionNum(SIGNER_NYM, strServerID, lNoticeNum, false); // bSave = false (but saved below...)
02558             }
02559 
02560             // We save regardless of whether any removals or additions are made, because data was
02561             // updated in UpdateHighestNum regardless.
02562             //
02563             if (bSave)
02564                 SaveSignedNymfile(SIGNER_NYM);
02565         }
02566     }
02567 }
