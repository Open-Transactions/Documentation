01032     //
01033     SetPurportedAccountID(theAcctID);

01742     /*

01744     -- Iterate through the transactions in the payments inbox.
01745     -- (They should all be "instrumentNotice" transactions.)
01746     -- Each transaction contains (1) OTMessage in "in ref to" field, which in turn contains an encrypted
01747     OTPayment in the payload field, which contains the actual financial instrument.
01748     -- *** Therefore, this function, based purely on ledger index (as we iterate):
01749     1. extracts the OTMessage from the Transaction "in ref to" field (for the transaction at that index),
01750     2. then decrypts the payload on that message, producing an OTPayment object,
01751     3. ...which contains the actual instrument.
01752     */

01755         (OTTransaction::payDividend         != pTransaction->GetType()) &&

01764         (OTTransaction::instrumentNotice == pTransaction->GetType()) || // It's encrypted.
01765         (OTTransaction::payDividend      == pTransaction->GetType())
01766         )

01770

01772         {

01774                 "contain an 'in reference to' string, but it was empty. (Returning \"\".)\n", __FUNCTION__);

01789         // Now we need to decrypt the payment on that message (which contains the instrument

01791         // OTAPI_Wrap::GetNym_MailContentsByIndex():
01792         //
01793 
01794         // SENDER:     pMsg->m_strNymID

01799         OTString    strEnvelopeContents;
01800 
01801         // Decrypt the Envelope.

01806             OTLog::vOutput(0, "%s: Failed trying to decrypt the financial instrument "
01807                            "that was supposedly attached as a payload to this payment message:\n%s\n\n",
01808                            __FUNCTION__, strMsg.Get());

01810                 OTLog::vOutput(0, "%s: Failed: after decryption, cleartext is empty. From:\n%s\n\n",
01811                                __FUNCTION__, strMsg.Get());
01812         else
01813         {
01814             OTPayment * pPayment = new OTPayment(strEnvelopeContents);  // strEnvelopeContents contains a PURSE or CHEQUE (etc) and not specifically a PAYMENT.
01815             OT_ASSERT(NULL != pPayment);
01816             OTCleanup<OTPayment> thePaymentAngel(pPayment);
01817 
01818             if (!pPayment->IsValid())
01819                 OTLog::vOutput(0, "%s: Failed: after decryption, payment is invalid. "

01821             else // success.
01822             {
01823                 thePaymentAngel.SetCleanupTargetPointer(NULL);
01824                 return pPayment; // Caller responsible to delete.
01825             }
01826         }
01827     }
01828     else
01829         OTLog::vError("%s: This must be a notice (vs an instrumentNotice or payDividend). "
01830                       "!!! Not yet supported !!!\n", __FUNCTION__);
01831 
01832     return NULL;
01833 }

01950                 default: // todo: possibly change this to an OT_ASSERT. security.
