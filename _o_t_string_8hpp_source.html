00245 /*
00246  * Copy src to string dst of size siz.  At most siz-1 characters
00247  * will be copied.  Always NUL terminates (unless siz == 0).
00248  * Returns strlen(src); if retval >= siz, truncation occurred.
00249  */
00250 inline size_t strlcpy(char *dst, const char *src, size_t siz)
00251 {
00252     char *d = dst;
00253     const char *s = src;
00254     size_t n = siz;
00255 
00256     /* Copy as many bytes as will fit */
00257     if (n != 0)
00258     {
00259         while (--n != 0)
00260         {
00261             if ((*d++ = *s++) == '\0')
00262                 break;
00263         }
00264     }
00265 
00266     /* Not enough room in dst, add NUL and traverse rest of src */
00267     if (n == 0)
00268     {
00269         if (siz != 0)
00270             *d = '\0';  /* NUL-terminate dst */
00271         while (*s++)
00272             ;
00273     }
00274 
00275     return(s - src - 1); /* count does not include NUL */
00276 }
00277 /*
00278  * Appends src to string dst of size siz (unlike strncat, siz is the
00279  * full size of dst, not space left).  At most siz-1 characters
00280  * will be copied.  Always NUL terminates (unless siz <= strlen(dst)).
00281  * Returns strlen(src) + MIN(siz, strlen(initial dst)).
00282  * If retval >= siz, truncation occurred.
00283  */
00284 inline size_t strlcat(char *dst, const char *src, size_t siz)
00285 {
00286     char *d = dst;
00287     const char *s = src;
00288     size_t n = siz;
00289     size_t dlen;
00290 
00291     /* Find the end of dst and adjust bytes left but don't go past end */
00292     while (n-- != 0 && *d != '\0')
00293         d++;
00294     dlen = d - dst;
00295     n = siz - dlen;
00296 
00297     if (n == 0)
00298         return(dlen + strlen(s));
00299     while (*s != '\0')
00300     {
00301         if (n != 1)
00302         {
00303             *d++ = *s;
00304             n--;
00305         }
00306         s++;
00307     }
00308     *d = '\0';
00309 
00310     return(dlen + (s - src)); /* count does not include NUL */
00311 }
