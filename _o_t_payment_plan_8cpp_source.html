01050                 
01051                 // (I do NOT save m_pCron here, since that already occurs after this function is called.)
01052             }
01053             else // bSuccess = false.  The payment failed.
01054             {
01055                 pItemSend ->SetStatus(OTItem::rejection);// pSourceAcct     // These are already initialized to false.
01056                 pItemRecip->SetStatus(OTItem::rejection);// pRecipientAcct  // (But just making sure...)

01058                 pItemSend ->SetAmount(0);    // No money changed hands. Just being explicit.
01059                 pItemRecip->SetAmount(0);    // No money changed hands. Just being explicit.

01072                 }
01073             }

01076             // CURRENT copy of the payment plan (which took just money from the user's acct, or not,
01077             // and either way thus updated its status -- so its internal data has changed.)
01078             //
01079             // It will also contain a copy of the user's ORIGINAL signed payment plan, where the data
01080             // has NOT changed, (so the user's original signature is still good.)
01081             //
01082             // In order for it to export the RIGHT VERSION of the CURRENT plan, which has just changed
01083             // (above), then I need to re-sign it and save it first. (The original version I'll load from
01084             // a separate file using OTCronItem::LoadCronReceipt(lTransactionNum). It has both original
01085             // signatures on it. Nice, eh?)

01087             
01088             this->ReleaseSignatures();

01090             this->SaveContract();

01092             
01093             // No need to save Cron here, since both caller functions call SaveCron() EVERY time anyway,
01094             // success or failure, rain or shine.
01095             //m_pCron->SaveCron(); // Cron is where I am serialized, so if Cron's not saved, I'm not saved.
01096             
01097             // -----------------------------------------------------------------
01098             //
01099             // EVERYTHING BELOW is just about notifying the users, by dropping the receipt in their
01100             // inboxes. The rest is done.  The accounts and inboxes will all be saved at the same time.
01101             //
01102             // The Payment Plan is entirely updated and saved by this point, and Cron will
01103             // also be saved in the calling function once we return (no matter what.)
01104             //
01105             // -----------------------------------------------------------------
01106             
01107             
01108             // Basically I load up both INBOXES, which are actually LEDGERS, and then I create
01109             // a new transaction, with a new transaction item, for each of the ledgers.
01110             // (That's where the receipt information goes.)
01111             // 
01112             
01113             
01114             
01115             // -----------------------------------------------------------------
01116             
01117             // The TRANSACTION will be sent with "In Reference To" information containing the
01118             // ORIGINAL SIGNED PLAN. (With both of the users' original signatures on it.)
01119             //
01120             // Whereas the TRANSACTION ITEM will include an "attachment" containing the UPDATED
01121             // PLAN (this time with the SERVER's signature on it.)
01122             
01123             // (Lucky I just signed and saved the updated plan (above), or it would still have 
01124             // have the old data in it.)
01125             
01126             // I also already loaded the original plan. Remember this from above,
01127             // near the top of the function:
01128             //  OTCronItem * pOrigCronItem  = NULL;
01129             //  OTString strOrigPlan(*pOrigCronItem); // <====== Farther down in the code, I attach this string to the receipts.
01130             //  ... then lower down...
01131             // pTransSend->SetReferenceString(strOrigPlan);
01132             // pTransRecip->SetReferenceString(strOrigPlan);
01133             //
01134             // So the original plan is already loaded and copied to the Transaction as the "In Reference To" 
01135             // Field. Now let's add the UPDATED plan as an ATTACHMENT on the Transaction ITEM:
01136             //
01137             OTString    strUpdatedPlan(*this);
01138             
01139             // Set the updated plan as the attachment on the transaction item.
01140             // (With the SERVER's signature on it!)
01141             // (As a receipt for each trader, so they can see their offer updating.)
01142             pItemSend->SetAttachment(strUpdatedPlan);
01143             pItemRecip->SetAttachment(strUpdatedPlan);          
01144             // -----------------------------------------------------------------
01145             // Success OR failure, either way I want a receipt in both inboxes.
01146             // But if FAILURE, I do NOT want to save the Accounts, JUST the inboxes.
01147             // So the inboxes happen either way, but the accounts are saved only on success.
01148             
01149             // sign the item
01150             pItemSend->SignContract(*pServerNym);
01151             pItemRecip->SignContract(*pServerNym);
01152             
01153             pItemSend->SaveContract();
01154             pItemRecip->SaveContract();
01155             
01156             // the Transaction "owns" the item now and will handle cleaning it up.
01157             pTransSend->AddItem(*pItemSend);
01158             pTransRecip->AddItem(*pItemRecip);
01159             
01160             pTransSend->SignContract(*pServerNym);
01161             pTransRecip->SignContract(*pServerNym);
01162             
01163             pTransSend->SaveContract();
01164             pTransRecip->SaveContract();
01165             // -------------------------------------------
01166             // Here, the transactions we just created are actually added to the ledgers.
01167             // This happens either way, success or fail.
01168             
01169             theSenderInbox.     AddTransaction(*pTransSend);
01170             theRecipientInbox.  AddTransaction(*pTransRecip);
01171             // -------------------------------------------
01172             // Release any signatures that were there before (They won't
01173             // verify anymore anyway, since the content has changed.)
01174             theSenderInbox.     ReleaseSignatures();
01175             theRecipientInbox.  ReleaseSignatures();
01176             
01177             // Sign both of them.               
01178             theSenderInbox.     SignContract(*pServerNym);
01179             theRecipientInbox.  SignContract(*pServerNym);
01180             
01181             // Save both of them internally
01182             theSenderInbox.     SaveContract();
01183             theRecipientInbox.  SaveContract();
01184                         
01185             // Save both inboxes to storage. (File, DB, wherever it goes.)
01186             pSourceAcct->       SaveInbox(theSenderInbox);  
01187             pRecipientAcct->    SaveInbox(theRecipientInbox);   
01188 
01189             // These correspond to the AddTransaction() calls just above. These are stored
01190             // in separate files now.
01191             //
01192             pTransSend ->SaveBoxReceipt(theSenderInbox);
01193             pTransRecip->SaveBoxReceipt(theRecipientInbox);
01194             
01195             // If success, save the accounts with new balance. (Save inboxes with receipts either way,
01196             // and the receipts will contain a rejection or acknowledgment stamped by the Server Nym.)
01197             if (true == bSuccess)
01198             {                   
01199                 // -------------------------------------------
01200                 // Release any signatures that were there before (They won't
01201                 // verify anymore anyway, since the content has changed.)
01202                 pSourceAcct->   ReleaseSignatures();    
01203                 pRecipientAcct->ReleaseSignatures();    
01204 
01205                 // Sign both of them.               
01206                 pSourceAcct->   SignContract(*pServerNym);  
01207                 pRecipientAcct->SignContract(*pServerNym);  
01208 
01209                 // Save both of them internally
01210                 pSourceAcct->   SaveContract(); 
01211                 pRecipientAcct->SaveContract(); 
01212 
01213                 // TODO: Better rollback capabilities in case of failures here:
01214                 
01215                 // Save both accounts to storage.
01216                 pSourceAcct->   SaveAccount();  
01217                 pRecipientAcct->SaveAccount();  
01218                 
01219                 // NO NEED TO LOG HERE, since success / failure is already logged above.
01220             }
01221         } // both inboxes were successfully loaded or generated.
01222     } // By the time we enter this block, accounts and nyms are already loaded. As we begin, inboxes are instantiated.
01223     
01224     return bSuccess;
01225 }

01338     OT_ASSERT(NULL != GetCron());
01339     
01340     // -----------------------------------------------------------------
01341     // Right now Cron is called 10 times per second.
01342     // I'm going to slow down all trades so they are once every GetProcessInterval()
01343     if (GetLastProcessDate() > OT_TIME_ZERO)
01344     {
01345 //      OTLog::vOutput(3, "DEBUG: time: %d  Last process date: %d   Time since last: %d    Interval: %d\n",
01346 //                     OTTimeGetCurrentTime(), GetLastProcessDate(), (OTTimeGetCurrentTime() - GetLastProcessDate()),
01347 //                     GetProcessInterval());

01349         // (Default ProcessInternal is 1 second, but Trades will use 10 seconds,
01350         // and Payment Plans will use an hour or day.)
01351         if (OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetLastProcessDate()) <= GetProcessInterval())
01352             return true;
01353     }
01354     // Keep a record of the last time this was processed.
01355     // (NOT saved to storage, only used while the software is running.)
01356     // (Thus no need to release signatures, sign contract, save contract, etc.)
01357     SetLastProcessDate(OTTimeGetCurrentTime());
01358     // -----------------------------------------------------------------
01359             
01360     // END DATE --------------------------------
01361     // First call the parent's version (which this overrides) so it has
01362     // a chance to check its stuff. 
01363     // Currently it calls OTCronItem::ProcessCron, which checks IsExpired().
01364     //
01365     if (false == ot_super::ProcessCron())
01366     {
01367         OTLog::Output(3, "Cron job has expired.\n");
01368         return false;   // It's expired or flagged for removal--remove it from Cron.
01369     }
01370 
01371     // START DATE --------------------------------
01372     // Okay, so it's not expired. But might not have reached START DATE yet...
01373     if (!VerifyCurrentDate())
01374         return true;    // The Payment Plan is not yet valid, so we return. BUT, we also 
01375                         // return TRUE, so it will STAY on Cron until it BECOMES valid.
01376     
01377     
01378     // -----------------------------------------------------------------------------
01379 
01380     if (GetCron()->GetTransactionCount() < 1)
01381     {
01382         OTLog::Output(0, "Failed to process payment: Out of transaction numbers!\n");
01383         return true;    // If there aren't enough transaction numbers, this won't log
01384                         // 10 times per second, but instead every hour or every day, 
01385     }                   // since plans don't process any more often than that anyway.
01386     
01387     // -----------------------------------------------------------------------------
01388 
01389     // First process the initial payment...
01390     
01391     if (HasInitialPayment()                             &&  // If I have an initial payment...
01392         !IsInitialPaymentDone()                         &&  // and I have not yet processed it...
01393         (OTTimeGetCurrentTime() > GetInitialPaymentDate())  &&  // and we're past the initial payment due date...
01394         (OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetLastFailedInitialPaymentDate()) > OTTimeGetSecondsFromTime(OT_TIME_DAY_IN_SECONDS))) // and it's been more than a day since I last failed attmpting this...
01395     {   // THEN we're due for the initial payment! Process it!
01396 
01397         OTLog::Output(3, "Cron: Processing initial payment...\n");
01398 
01399         ProcessInitialPayment();
01400     }
01401     
01402     // -----------------------------------------------------------------------------
01403 
01404     // Next, process the payment plan...
01405     OTLog::vOutput(3, "(payment plan): Flagged/Removal: %s Has Plan: %s Current time: %" PRId64" Start Date: %" PRId64"\n",
01406                    (IsFlaggedForRemoval() ? "TRUE" : "FALSE"), (HasPaymentPlan() ? "TRUE" : "FALSE"), 
01407                    OTTimeGetSecondsFromTime(OTTimeGetCurrentTime()), OTTimeGetSecondsFromTime(GetPaymentPlanStartDate()));
01408     
01409     if (!IsFlaggedForRemoval() && HasPaymentPlan() &&       // This object COULD have gotten flagged for removal during the ProcessInitialPayment()
01410         (OTTimeGetCurrentTime() > GetPaymentPlanStartDate()))       // call. Therefore, I am sure to check that it's NOT IsFlaggedForRemoval() before calling
01411     {                                                       // this block of code.
01412 //      OTLog::Error("DEBUG: Payment Plan -------------\n");
01413         
01414         // First I'll calculate whether the next payment would be due, based on start date,
01415         // time between payments, and date of last payment.
01416         
01417         const int64_t DURATION_SINCE_START = OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetPaymentPlanStartDate());
01418         
01419         // Let's say the plan charges every week, and it's been 16 DAYS DURATION since starting.
01420         // The first charge would have been on the 1st day, 16 days ago.
01421         // Then the second charge would have been on the 8th day, (7 days later)
01422         // Then the third charge would have been on the 15th day, (7 days later again)
01423         // That means the next charge isn't due until the 22nd.
01424         
01425         // Right now in this example, DURATION_SINCE_START is: (16 * OT_TIME_DAY_IN_SECONDS).
01426         // I must calculate from that, that three charges have already happened, and that the
01427         // next one is not yet due.
01428         // 
01429         // I also know that GetTimeBetweenPayments() is set to (OT_TIME_DAY_IN_SECONDS * 7)
01430         // 
01431         // Duration / timebetween == 16/7 == 2 with 2 remainder.    (+1 to get 3: THREE should have already happened.)
01432         // if it was the 14th, 14/7 == 2 with 0 remainder.          (+1 to get 3: THREE should have happened by the 14th)
01433         // If it was the 22nd, 22/7 == 3 with 1 remainder.          (+1 to get 4: FOUR payments should have already happened.)
01434         //
01435         // Can also just add the TimeBetweenPayments to the DateOfLastPayment...
01436         //
01437         const int64_t nNoPaymentsThatShouldHaveHappenedByNow = DURATION_SINCE_START / OTTimeGetSecondsFromTime(GetTimeBetweenPayments()) + 1;
01438         // The +1 is because it charges on the 1st day of the plan. So 14 days, which is 7 times 2, equals *3* payments, not 2.
01439 
01440 //      OTLog::vOutput(3, "Payments that should have happened by now: %d\n"
01441 //                     "Number payments done: %d      date of last payment: %d\n"
01442 //                     "Date of last failed payment: %d   Time Between: %d", 
01443 //                     nNoPaymentsThatShouldHaveHappenedByNow, GetNoPaymentsDone(), GetDateOfLastPayment(),
01444 //                     GetDateOfLastFailedPayment(), GetTimeBetweenPayments());
01445         // -------------------------------------------------------
01446         
01447         // It's expired, remove it. (I check >0 because this one is an optional field.)
01448         if ((GetMaximumNoPayments() > 0) && (GetNoPaymentsDone() >= GetMaximumNoPayments()))
01449         {
01450             OTLog::Output(1, "Payment plan has expired by reaching max number of payments allowed.\n");
01451             return false; // This payment plan will be removed from Cron by returning false.
01452         }
01453         // Again, I check >0 because the plan length is optional and might just be 0.
01454         else if ((GetPaymentPlanLength() > OT_TIME_ZERO) && (OTTimeGetCurrentTime() >= OTTimeAddTimeInterval(GetPaymentPlanStartDate(), OTTimeGetSecondsFromTime(GetPaymentPlanLength()))))
01455         {
01456             OTLog::Output(1, "Payment plan has expired by reaching its maximum length of time.\n");
01457             return false; // This payment plan will be removed from Cron by returning false.
01458         }
01459         else if (nNoPaymentsThatShouldHaveHappenedByNow <=  GetNoPaymentsDone())    // if not enough payments have happened...
01460         {
01461 //          OTLog::Output(3, "DEBUG: Enough payments have already been made.\n");
01462         }
01463         else if (OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetDateOfLastPayment()) < OTTimeGetSecondsFromTime(GetTimeBetweenPayments())) // and the time since last payment is more than the payment period...
01464         {
01465 //          OTLog::Output(3, "DEBUG: Not enough time has elapsed.\n");  
01466         }
01467         else if (OTTimeGetTimeInterval(OTTimeGetCurrentTime(), GetDateOfLastFailedPayment()) < OTTimeGetSecondsFromTime(OT_TIME_DAY_IN_SECONDS)) // and it's been at least 24 hrs since the last failed payment...
01468         {
01469             OTLog::Output(3, "Cron (processing payment plan): Not enough time since last failed payment.\n");
01470         }
01471         // Okay -- PROCESS IT!
01472         else                    // The above 3 end-comments have opposite logic from their if(), since they used to be here.
01473         {                       // I reversed the operators so they could be failures, resulting in this else block for success.
01474             
01475             OTLog::Output(3, "Cron: Processing payment...\n");
01476             
01477             // This function assumes the payment is due, and it only fails in the case of 
01478             // the payer's account having insufficient funds.
01479             ProcessPaymentPlan();
01480         }
01481     }
01482     
01483 
01484     // Notice something: Markets are very concerned whether a trade failed, or if the account
01485     // was short of funds. They track that, and remove any trades when they have this problem.
01486     // So in OTTrade right now, you would be checking if it was flagged for removal, and 
01487     // returning false in that case.
01488     // 
01489     // But with a PAYMENT PLAN, if a payment fails, you don't want to cancel the plan!!!
01490     // You want it to keep trying until it gets in more payments, and ONLY cancel in the
01491     // case where the user REQUESTS it, or when one of the legitimate terms above expires naturally.
01492     // Insufficient funds? NO PROBLEM: you can stay on your payment plan as long as you want! :-)
01493     //
01494     // There ARE however funny cases where you WOULD want the plan removed.
01495     // For example:
01496     //
01497     if (IsFlaggedForRemoval() ||
01498         (HasInitialPayment() && IsInitialPaymentDone() && !HasPaymentPlan()))
01499     {
01500         OTLog::Output(3, "OTPaymentPlan::ProcessCron: Removing payment plan from cron processing...\n");
01501         return false; // if there's no plan, and initial payment is done, nothing left to do. Remove!
01502     }   
01503 
01504     return true;
01505 }
